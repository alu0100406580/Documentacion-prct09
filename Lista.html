def insertaAtras(valor)

  nodo = Node.new(valor,nil,nil)
  if (self.tamaño == 0)
    @cabeza = nodo
    @cola = nodo
  else
    nodo["prev"] = @cola
    @cola.next = nodo
    @cola = nodo
  end
  @tamaño = @tamaño + 1
end

def to_s
  nodoAux = @cabeza
  arr = []
  if (self.tamaño == 0)
      return "Lista Vacía"
  else
      while(nodoAux != nil)                
          arr.push nodoAux["value"]
          nodoAux = nodoAux["next"]
      end
      return arr
  end
